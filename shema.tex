\tikzstyle{decision} = [diamond, draw, fill=blue!20,

\begin{tikzpicture}[node distance = 4cm, auto]
% Place nodes
\node [cloud] (init) {begin};

\node [decision, below of=input] (if) { $ ((\sqrt(a) \leq b) \&\& (a \leq c)) $ };
\node [io, right of= if] (output) {write( a )};

\node [block, below  of= do2] (do3) {b = c};
\node [cloud, below of=output, node distance=3cm] (stop) {stop};
% Draw edges
\path [line] (init) -- (input);
\path [line] (input) -- (if);

\path [line] (do1) -- (do2);
\path [line] (do2) -- (do3);
